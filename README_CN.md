```java
// 渐进式恐惧公式
if (height <= 80) {
    damage = 0.0F;  // 安全地带
} else if (height <= 120) {
    damage = (height - 80) * 0.1F;  // 每层+0.05♥
} else {
    damage = 4.0F;  // 极度恐高：2♥
}
```
